function add2(a as int, b as int) as int
    a + b;
end function

function sub2(a as int, b as int) as int
    a - b;
end function

function mul2(a as int, b as int) as int
    a * b;
end function

function max2(a as int, b as int) as int
    if a >= b then
        a;
    else
        b;
    end if
end function

function abs1(x as int) as int
    if x < 0 then
        0 - x;
    else
        x;
    end if
end function
