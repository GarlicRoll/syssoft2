function parse_number() as int
    dim ch as int
    dim num as int
    
    num = 0;
    ch = input();
    
    num = ch - 48;
    
    while true
        ch = input();
        if ch >= 48 then
            if ch <= 57 then
                num = num * 10 + (ch - 48);
            end if
        else
            break
        end if
    wend
    
    num;
end function

function parse_factor() as int
    dim ch as int
    dim result as int
    
    ch = input();
    
    if ch == 40 then  
        result = parse_expression();
        ch = input(); 
        result;
    else
        if ch >= 48 then
            if ch <= 57 then
                result = ch - 48;

                while true
                    ch = input();
                    if ch >= 48 then
                        if ch <= 57 then
                            result = result * 10 + (ch - 48);
                        else
                            break
                        end if
                    else
                        break
                    end if
                wend
            else
                result = 0;
            end if
        else 
            result = 0;
        end if
        
        result;
    end if
end function

function parse_term() as int
    dim result as int
    dim op as int
    dim next_val as int
    dim saved_char as int
    result = parse_factor();
    
    while true
        op = input();
        
        if op == 42 then 
            next_val = parse_factor();
            result = result * next_val;
        else
            if op == 47 then  
                next_val = parse_factor();
                result = result / next_val;
            else
                saved_char = op;
                break
            end if
        end if
    wend
    
    result;
end function

function parse_expression() as int
    dim result as int
    dim op as int
    dim next_val as int
    
    result = parse_term();
    
    while true
        op = input();
        
        if op == 43 then 
            next_val = parse_term();
            result = result + next_val;
        else
            if op == 45 then 
                next_val = parse_term();
                result = result - next_val;
            else
                if op == 13 then  
                    break
                else
                    if op == 10 then
                        break
                    else
                        break
                    end if
                end if
            end if
        end if
    wend
    
    result;
end function

function number_length(num as int) as int
    dim rest as int
    dim count as int
    count = 1;
    rest = num/10;
    while rest != 0
        rest = rest/10;
        count = count + 1;
    wend
    count;
end function
