function fibonacci(n as int) as int
    if n <= 1 then
        n;
    else
        fibonacci(n - 1) + fibonacci(n - 2);
    end if
end function

function read_num() as int
    dim in as int
    dim res as int
    dim i as int
    res = 0;
    i = 1;
    while true
        in = input();
        if in == 13 then
            input();
            break
        end if
        res = res * 10 + (in - 48);
        i = i + 1;
    wend
    res;
end function

function number_length(num as int) as int
    dim rest as int
    dim count as int
    count = 1;
    rest = num/10;
    while rest != 0
        rest = rest/10;
        count = count + 1;
    wend
    count;
end function

function print_num(num as int) as int
    dim rest as int
    dim count as int
    dim shift as int
    rest = num;
    count = number_length(num) - 1;
    shift = 1;
    while count != 0
        shift = shift * 10;
        count = count - 1;
    wend
    while shift != 0
        output(rest / shift + 48);
        rest = rest % shift;
        shift = shift / 10;
    wend
    output(10);
    output(13);
    0;
end function

    dim n as int
    dim result as int

    
    n = read_num();
    
    while n >= 0
        result = fibonacci(n - 1);
