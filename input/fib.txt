function fibonacci(n as int) as int
    if n <= 1 then
        n;
    else
        fibonacci(n - 1) + fibonacci(n - 2);
    end if
end function

function read_num() as int
    dim in as int
    dim res as int
    dim i as int
    dim started as int
    dim sign as int
    res = 0;
    i = 1;
    started = 0;
    sign = 1;
    while true
        in = input();
        if in == 13 then
            if started == 1 then
                break
            end if
        end if
        if in == 10 then
            if started == 1 then
                break
            end if
        end if
        if in >= 48 then
            if in <= 57 then
                res = res * 10 + (in - 48);
                i = i + 1;
                started = 1;
            else
                if started == 1 then
                    break
                end if
            end if
        else
            if started == 0 then
                if in == '-' then
                    sign = 0 - 1;
                end if
            else
                break
            end if
        end if
    wend
    res * sign;
end function

function number_length(num as int) as int
    dim rest as int
    dim count as int
    count = 1;
    rest = num/10;
    while rest != 0
        rest = rest/10;
        count = count + 1;
    wend
    count;
end function

function print_num(num as int) as int
    dim rest as int
    dim count as int
    dim shift as int
    rest = num;
    count = number_length(num) - 1;
    shift = 1;
    while count != 0
        shift = shift * 10;
        count = count - 1;
    wend
    while shift != 0
        output(rest / shift + 48);
        rest = rest % shift;
        shift = shift / 10;
    wend
    output(10);
    output(13);
    0;
end function

function main()
    dim n as int
    dim result as int

    
    n = read_num();
    
    while n >= 0
        result = fibonacci(n - 1);
        
        print_num(result);
                
        n = read_num();
    wend
    

    0;
end function
