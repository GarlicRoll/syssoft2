function print_a(arg as int) as int
    while 1
        print(65);
    wend
end function

function print_b(arg as int) as int
    while 1
        print(66);
    wend
end function

function main() as int
    dim qOpsOk as int
    dim qTicksOk as int
    dim timerOk as int
    dim print_a as int
    dim print_b as int

    threadingInit();
    qOpsOk = setThreadQuantumOps(1);
    qTicksOk = setThreadQuantumTicks(1);
    timerOk = setupEntropyTimer(1);

    spawnThread(0, print_a);
    spawnThread(1, print_b);

    while 1
        0;
    wend
end function
