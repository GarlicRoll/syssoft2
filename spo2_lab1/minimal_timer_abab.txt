function t1(arg as int) as int
    while 1
        print(65);
    wend
end function

function t2(arg as int) as int
    while 1
        print(66);
    wend
end function

function main() as int
    dim qOpsOk as int
    dim qTicksOk as int
    dim timerOk as int
    dim t1 as int
    dim t2 as int

    threadingInit();
    qOpsOk = setThreadQuantumOps(1);
    qTicksOk = setThreadQuantumTicks(1);
    timerOk = setupEntropyTimer(1);

    spawnThread(0, t1);
    spawnThread(1, t2);

    startThreadScheduler();
end function
